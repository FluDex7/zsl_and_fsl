\section*{Контрастивный Loss (InfoNCE)}

Батч из $N$ пар. Матрица сходства $N\times N$:

\[
\text{logits}_{ij} = \cos(\mathbf{I_e}^{(i)}, \mathbf{T_e}^{(j)}) \cdot \exp(\tau)
\]

\[
\boxed{
\mathcal{L} = \frac{1}{2}\left[
  \underbrace{\mathrm{CE}(\text{logits}, \text{labels}, \text{rows})}_{\text{image} \to \text{text}}
  + \underbrace{\mathrm{CE}(\text{logits}, \text{labels}, \text{cols})}_{\text{text} \to \text{image}}
\right]
}
\]

$\text{labels} = [0, 1, \ldots, N\!-\!1]$, \quad $\tau$ --- обучаемый скаляр.

\bigskip

\noindent\textbf{Псевдокод (из статьи):}
\begin{verbatim}
  I_f = image_encoder(I)           # [n, d_i]
  T_f = text_encoder(T)            # [n, d_t]
  I_e = l2_normalize(I_f @ W_i)    # [n, d_e]
  T_e = l2_normalize(T_f @ W_t)    # [n, d_e]
  logits = I_e @ T_e.T * exp(t)    # [n, n]
  labels = arange(n)
  loss_i = cross_entropy(logits, labels, axis=0)
  loss_t = cross_entropy(logits, labels, axis=1)
  loss   = (loss_i + loss_t) / 2
\end{verbatim}


\section*{CLIP vs SigLIP: отличие в Loss}

\begin{center}
\begin{tabular}{p{2.5cm}p{5.5cm}p{5.5cm}}
\toprule
& \textbf{CLIP (Softmax / InfoNCE)} & \textbf{SigLIP (Sigmoid)} \\
\midrule
\textbf{Формула} &
  $\mathrm{CE}(\mathrm{softmax}(\text{logits}), \text{labels})$ &
  $-\sum_{ij} \log \sigma(z_{ij} \cdot s_{ij})$ \\[6pt]
\textbf{Нормализация} &
  Глобальная (по всему батчу) &
  Независимая (каждая пара) \\[6pt]
\textbf{Батч} &
  Нужен большой (32K) &
  Работает при любых размерах \\[6pt]
\textbf{ImageNet ZS} &
  76.2\% (ViT-L/14@336) &
  Превосходит CLIP \\
\bottomrule
\end{tabular}
\end{center}

\noindent$z_{ij} = +1$ если пара совпадает, $-1$ иначе.\quad
$s_{ij} = \tau \cdot \cos(\mathbf{I_e}^{(i)}, \mathbf{T_e}^{(j)}) + b$.
